

 ## 01.Positive/Negative

````java[]

import java.util.*;
public class Main{
    public static void main(String[]args){
        int num = 345;
        if(num>0){
            System.out.println("Positive number");
        }
        else{
            System.out.println("negative number");
        }
    }
}
````

 ## 02.Odd/Even 

````java[]

import java.util.*;
public class Main{
    public static void main(String[]args){
        int num = 357;
        if(num %2 ==0){
            System.out.println("even number");
        }
        else{
            System.out.println("odd number");
        }
    }
    
}
````
